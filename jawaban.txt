# Jawaban soal nomer 2
a. Table users :
CREATE TABLE users (id int auto_increment primary key, name varchar(255), email varchar(255), password(255));

b. Table categories :
CREATE TABLE categories (id int auto_increment primary key, name varchar(255));

c. Table items :
CREATE TABLE items (id int auto_increment primary key, name varchar(255), description varchar(255), price int, stock int, category_id int, foreign key(category_id) references categories(id));

# Jawaban soal nomer 4
a. Mengambil data users kecuali password :
SELECT id, name, email FROM users;

b. Mengambil data items diatas harga 1 juta dan memiliki name serupa(like) dengan keyword 'sang' :
SELECT * FROM items WHERE price >= 1000000;
SELECT * FROM items WHERE name LIKE '%sang%';

c. Menampilkan data items join dengan kategori :
SELECT items.name, items.description, items.price, items.stock, items.category_id, categories.name FROM items INNER JOIN categories ON items.category_id = categories.id;
